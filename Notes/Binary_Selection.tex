% Slide 1: Practical Example
\begin{frame}{Practical Application Example: Spam Email Classification}
		\begin{exampleblock}{Example Calculation}
			For email containing "free", "prize":
			\begin{itemize}
				\item Output scores: [2.3, 5.1] (normal vs spam)
				\item Calculations:
				\begin{align*}
					e^{2.3} &\approx 9.9742 \\
					e^{5.1} &\approx 164.022 \\
					P0 &= \frac{9.9742}{9.9742+164.022} \approx 0.0573 \\
					P1 &= \frac{164.022}{9.9742+164.022} \approx 0.9427
				\end{align*}
				\item Decision: Spam (P1 > 0.5)
			\end{itemize}
		\end{exampleblock}
	\end{frame}
	
	% Slide 2: Network Architecture
	\begin{frame}{Network Architecture}
		
		\begin{block}{Core Features}
			\begin{itemize}
				\item \textbf{Dual-output design}: Two neurons for two classes
				\item \textbf{Probability normalization}: $\sum P_i = 1$
				\item \textbf{Symmetric optimization}: Learns both decision boundaries
			\end{itemize}
		\end{block}
	\end{frame}
	
	% Slide 3: Mathematical Formulation
	\begin{frame}{Mathematical Formulation}
		\begin{block}{Softmax Function}
			For output scores $\mathbf{s} = [s_1, s_2]$:
			\[
			P(y=1) = \frac{e^{s_1}}{e^{s_1} + e^{s_2}} \quad P(y=2) = \frac{e^{s_2}}{e^{s_1} + e^{s_2}}
			\]
		\end{block}
		
		\begin{block}{Cross-Entropy Loss}
			\[
			L = -\sum_{c=1}^2 y_c \log(p_c)
			\]
			Where:
			\begin{itemize}
				\item $y$: One-hot encoded true label
				\item $p$: Predicted probability
			\end{itemize}
		\end{block}
	\end{frame}
	
	% Slide 4: Calculation Example 1
	\begin{frame}{Calculation Example: Correct Prediction}
		\begin{block}{Parameters}
			\begin{itemize}
				\item Output scores: [3.0, -1.0]
				\item True label: Class 0 (y=[1,0])
			\end{itemize}
		\end{block}

% Slide 4: Calculation Example 1
\begin{frame}{Calculation Example: Correct Prediction}
		\begin{exampleblock}{Step-by-Step Calculation}
			\begin{enumerate}
				\item Compute exponents:
				\begin{align*}
					e^{3.0} &\approx 20.0855 \\
					e^{-1.0} &\approx 0.3679
				\end{align*}
				
				\item Compute probabilities:
				\begin{align*}
					P(y=0) &= {20.0855}/{(20.0855 + 0.3679)} \approx 0.9820 \\
					P(y=1) &= {0.3679}/{(20.0855 + 0.3679)} \approx 0.0180
				\end{align*}
				
				\item Cross-entropy loss:
				\[
				L = -[1\times\log(0.9820) + 0\times\log(0.0180)] \approx 0.0182
				\]
			\end{enumerate}
		\end{exampleblock}
	\end{frame}
	
	% Slide 5: Calculation Example 2
	\begin{frame}{Calculation Example: Incorrect Prediction}
		\begin{block}{Parameters}
			\begin{itemize}
				\item Output scores: [-2.0, 1.0]
				\item True label: Class 0 (y=[1,0])
			\end{itemize}
		\end{block}
		
		\begin{exampleblock}{Step-by-Step Calculation}
			\begin{enumerate}
				\item Compute exponents:
				\begin{align*}
					e^{-2.0} \approx 0.1353,
					e^{1.0} \approx 2.7183
				\end{align*}
				\begin{align*}
					P(y=0) &= {0.1353}/({0.1353 + 2.7183}) \approx 0.0474 \\
					P(y=1) &= {2.7183}/{(0.1353 + 2.7183)} \approx 0.9526
				\end{align*}
				
				\item Cross-entropy loss:
				\[
				L = -[1\times\log(0.0474) + 0\times\log(0.9526)] \approx 3.0486
				\]
			\end{enumerate}
		\end{exampleblock}
	\end{frame}
